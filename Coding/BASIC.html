Programming Basics for Kids

Welcome to the fun world of programming! This guide will take you through the basic concepts you need to understand to start your journey in computer programming.

Introduction to Programming Concepts

Imagine you have a robot.
--
This robot can do lots of cool things.
--
Like play music, move around, or even tell jokes.
--
But here's the thing: the robot doesn't know how to do any of these things by itself.
--
It needs you to tell it exactly what to do.
--
Programming is like writing a list of instructions for the robot so it can understand what you want it to do.
--
These instructions are called "code".

Translators: Interpreter vs. Compiler

  • Interpreter: Translates code line by line and executes it immediately. (Like a student reading a manual line by line and building a LEGO castle as they go.)
  • Compiler: Translates the entire code at once into machine code before execution. (Like a student reading the entire manual once completely and then building the entire castle from start to finish.)

JavaScript:

const overlay = document.getElementById('overlay');
        const textInput = document.getElementById('textInput');
        const video = document.querySelector('video');
        const retryButton = document.getElementById('retryButton');
        const colorPalette = document.getElementById('colorPalette');
        let chosenColor = 'red';

        textInput.addEventListener('input', function() {
            overlay.textContent = this.value;
        });

        colorPalette.addEventListener('click', function(event) {
            if (event.target.classList.contains('color-box')) {
                chosenColor = event.target.getAttribute('data-color');
                video.style.boxShadow = `0 4px 8px ${chosenColor}`;
                overlay.style.textShadow = `2px 2px 4px ${chosenColor}`;
            }
        });

        video.addEventListener('click', function(event) {
            const rect = video.getBoundingClientRect();
            const x = event.clientX - rect.left;
            const y = event.clientY - rect.top;

            const dot = document.createElement('div');
            dot.classList.add('dot');
            dot.style.backgroundColor = chosenColor;
            dot.style.left = `${x - 5}px`; // Adjust for dot size
            dot.style.top = `${y - 5}px`;

            const watermark = document.createElement('div');
            watermark.classList.add('watermark');
            watermark.style.left = `${x + 10}px`; // Adjust for dot size and positioning
            watermark.style.top = `${y + 10}px`;

            // Check if rounding is enabled
            const roundingCheckbox = document.getElementById('roundingCheckbox');
            const displayX = roundingCheckbox.checked ? Math.round(x) : x;
            const displayY = roundingCheckbox.checked ? Math.round(y) : y;

            watermark.textContent = `(${displayX}, ${displayY})`;

            const videoContainer = document.querySelector('.video-container');
            videoContainer.appendChild(dot);
            videoContainer.appendChild(watermark);
        });

        // Save Image button functionality
        document.getElementById('saveButton').addEventListener('click', function() {
            const video = document.querySelector('video');
            const overlay = document.querySelector('.video-overlay');
            const dots = document.querySelectorAll('.dot');
            const watermarks = document.querySelectorAll('.watermark');

            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');

            // Set canvas dimensions to match video
            canvas.width = video.videoWidth;
            canvas.height = video.videoHeight;

            // Draw the current video frame onto the canvas
            ctx.drawImage(video, 0, 0, canvas.width, canvas.height);

            // Draw overlay text
            ctx.font = `${window.getComputedStyle(overlay).fontSize} ${window.getComputedStyle(overlay).fontFamily}`;
            ctx.fillStyle = window.getComputedStyle(overlay).color;
            const overlayX = parseInt(window.getComputedStyle(overlay).left);
            const overlayY = parseInt(window.getComputedStyle(overlay).top) + parseInt(window.getComputedStyle(overlay).fontSize);
            ctx.fillText(overlay.textContent, overlayX, overlayY);

            // Draw dots
            dots.forEach(dot => {
                const dotX = parseInt(dot.style.left) + (parseInt(getComputedStyle(dot).width) / 2);
                const dotY = parseInt(dot.style.top) + (parseInt(getComputedStyle(dot).height) / 2);
                ctx.beginPath();
                ctx.arc(dotX, dotY, parseInt(getComputedStyle(dot).width) / 2, 0, Math.PI * 2);
                ctx.fillStyle = getComputedStyle(dot).backgroundColor;
                ctx.fill();
            });

            // Draw watermarks
            watermarks.forEach(watermark => {
                ctx.font = `${window.getComputedStyle(watermark).fontSize} ${window.getComputedStyle(watermark).fontFamily}`;
                ctx.fillStyle = window.getComputedStyle(watermark).color;
                const watermarkX = parseInt(window.getComputedStyle(watermark).left);
                const watermarkY = parseInt(window.getComputedStyle(watermark).top) + parseInt(window.getComputedStyle(watermark).fontSize);
                ctx.fillText(watermark.textContent, watermarkX, watermarkY);
            });

            // Convert canvas to image and download
            const link = document.createElement('a');
            link.href = canvas.toDataURL('image/png');
            link.download = 'screenshot.png';
            link.click();
        });


        function startVideoStream() {
            navigator.mediaDevices.getUserMedia({ video: true })
            .then(function(stream) {
                video.srcObject = stream;
                retryButton.style.display = 'none';
            })
            .catch(function(error) {
                console.log("Error accessing camera: ", error);
                overlay.textContent = 'Camera access denied. Please allow camera access and click Retry.';
                retryButton.style.display = 'block';
            });
        }

        function requestCameraAccess() {
            startVideoStream();
        }

        // Initialize video stream on page load
        startVideoStream();

config Configuration object that sets up the game's settings, including size and physics. { type: Phaser.AUTO, width: 800, height: 600 }
type: Phaser.AUTO Tells Phaser to use WebGL if possible, otherwise use Canvas. Difference: Determines the rendering technology used by Phaser. WebGL allows for hardware-accelerated rendering, offering better performance and graphical capabilities compared to Canvas. Canvas is a 2D drawing API supported by all major browsers, but it may not perform as well with complex animations or graphics.
width: 800, height: 600 Sets the width and height of the game screen to 800x600 pixels.
backgroundColor: '#87CEEB' Sets the background color of the game to a sky blue color.
parent: 'gameCanvas' Links the game to the HTML element with the ID 'gameCanvas'.
physics Adds physics settings to the game, like gravity and collision detection. { default: 'arcade', arcade: { gravity: { y: 300 }, debug: false } }
scene Contains the functions for loading assets, creating objects, and updating the game. { preload: preload, create: create, update: update }
game The main Phaser game object that starts the game with the configuration settings.
player The main character in the game that the player controls.
cursors Stores the keyboard input for controlling the player.
preload Function to load game assets (images, sounds) before the game starts.
this.load.image Loads an image asset for the game. this.load.image('sky', 'https://phaser.io/images/img.png')
this.load.spritesheet Loads a spritesheet, which is an image containing multiple frames of animation. this.load.spritesheet('dude', 'https://phaser.io/images/img.png', { frameWidth: 32, frameHeight: 48 })

    const config = {
            type: Phaser.AUTO,
            width: 800,
            height: 600,
            backgroundColor: '#87CEEB',
            parent: 'gameCanvas',
            physics: {
                default: 'arcade',
                arcade: {
                    gravity: { y: 300 },
                    debug: false
                }
            },
            scene: {
                preload: preload,
                create: create,
                update: update
            }
        };

        const game = new Phaser.Game(config);

        let player;
        let cursors;

        function preload() {
            this.load.image('sky', 'https://labs.phaser.io/assets/skies/space3.png');
            this.load.image('ground', 'https://labs.phaser.io/assets/sprites/platform.png');
            this.load.image('star', 'https://labs.phaser.io/assets/sprites/star.png');
            this.load.image('bomb', 'https://labs.phaser.io/assets/sprites/enemy-bullet.png'); // Corrected URL
            this.load.spritesheet('dude', 
                'https://labs.phaser.io/assets/sprites/dude.png', 
                { frameWidth: 32, frameHeight: 48 }
            );
        }

        function create() {
            this.add.image(400, 300, 'sky');

            const platforms = this.physics.add.staticGroup();

            platforms.create(400, 568, 'ground').setScale(2).refreshBody();
            platforms.create(600, 400, 'ground');
            platforms.create(50, 250, 'ground');
            platforms.create(750, 220, 'ground');

            player = this.physics.add.sprite(100, 450, 'dude');
            player.setBounce(0.2);
            player.setCollideWorldBounds(true);

            this.anims.create({
                key: 'left',
                frames: this.anims.generateFrameNumbers('dude', { start: 0, end: 3 }),
                frameRate: 10,
                repeat: -1
            });

            this.anims.create({
                key: 'turn',
                frames: [ { key: 'dude', frame: 4 } ],
                frameRate: 20
            });

            this.anims.create({
                key: 'right',
                frames: this.anims.generateFrameNumbers('dude', { start: 5, end: 8 }),
                frameRate: 10,
                repeat: -1
            });

            const stars = this.physics.add.group({
                key: 'star',
                repeat: 11,
                setXY: { x: 12, y: 0, stepX: 70 }
            });

            stars.children.iterate(function (child) {
                child.setBounceY(Phaser.Math.FloatBetween(0.4, 0.8));
            });

            this.physics.add.collider(player, platforms);
            this.physics.add.collider(stars, platforms);
            this.physics.add.overlap(player, stars, collectStar, null, this);

            cursors = this.input.keyboard.createCursorKeys();
        }

        function update() {
            if (cursors.left.isDown) {
                player.setVelocityX(-160);
                player.anims.play('left', true);
            } else if (cursors.right.isDown) {
                player.setVelocityX(160);
                player.anims.play('right', true);
            } else {
                player.setVelocityX(0);
                player.anims.play('turn');
            }

            if (cursors.up.isDown && player.body.touching.down) {
                player.setVelocityY(-330);
            }
        }
